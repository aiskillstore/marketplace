# Network Configuration Enumeration
# Usage: python3 telnet_helper.py --host IP --port PORT --script enum_network.txt

# Network interfaces
ifconfig -a
cat /proc/net/dev

# Routing table
route -n
cat /proc/net/route

# DNS configuration
cat /etc/resolv.conf
cat /etc/hosts

# Network connections and listening ports
netstat -tulpn
netstat -an

# ARP table
arp -a
cat /proc/net/arp

# Wireless configuration (if applicable)
iwconfig
cat /proc/net/wireless

# Firewall rules
iptables -L -n -v

# Network statistics
netstat -s
cat /proc/net/snmp
