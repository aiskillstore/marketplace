# File System Exploration
# Usage: python3 telnet_helper.py --host IP --port PORT --script enum_files.txt

# Root directory listing
ls -la /

# Important directories
ls -la /etc
ls -la /tmp
ls -la /var
ls -la /home
ls -la /root

# Web server directories (if present)
ls -la /var/www
ls -la /usr/share/nginx
ls -la /srv

# Configuration files
ls -la /etc/*.conf
ls -la /etc/config

# Init scripts
ls -la /etc/init.d
ls -la /etc/rc.d

# Binary directories
ls -la /bin
ls -la /sbin
ls -la /usr/bin
ls -la /usr/sbin

# Library directories
ls -la /lib
ls -la /usr/lib

# Device information
ls -la /dev

# Proc filesystem interesting files
cat /proc/sys/kernel/hostname
cat /proc/sys/kernel/version
